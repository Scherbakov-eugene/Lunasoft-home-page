package yourprojectpackage.gui; 
 
import yourprojectpackage.resources.PropertyManager; 
 
import javax.swing.*; 
import java.awt.*; 
import java.awt.event.ActionEvent; 
import java.awt.event.ActionListener; 
import java.util.HashMap; 
import java.util.Properties; 
import java.util.Vector; 
 
public class LAFManager 
{ 
    private static JMenu menu; 
    private static HashMap<String, String> lafNameAndClassName = new HashMap<String, String>(); 
 
    private static Vector<Component> listeners = new Vector<Component>(); 
 
    public static void addUIListener(Component listener) 
    { 
        listeners.add(listener); 
    } 
 
    public static void removeUIListener(Component listener) 
    { 
        listeners.remove(listener); 
    } 
 
    public static JMenu getLAFMenu() 
    { 
        if (menu == null) 
        { 
            menu = new JMenu("Look And Feel"); 
 
            // Get standard LAF list 
            UIManager.LookAndFeelInfo[] lafs = UIManager.getInstalledLookAndFeels(); 
 
            // Standard LAFs 
            for (UIManager.LookAndFeelInfo laf : lafs) 
            { 
                menu.add(getLAFMenuItem(laf.getName(), laf.getClassName())); 
            } 
            menu.addSeparator(); 
 
            // Extended LAFs 
            menu.add(getLAFMenuItem("3D LAF v2.7", "swing.addon.plaf.threeD.ThreeDLookAndFeel")); 
            menu.add(getLAFMenuItem("GTK+ v0.1", "org.gtk.java.swing.plaf.gtk.GtkLookAndFeel")); 
            menu.add(getLAFMenuItem("InfoNode LAF v1.5.0", "net.infonode.gui.laf.InfoNodeLookAndFeel")); 
            menu.add(getLAFMenuItem("Liquid LAF v2.9.1", "com.birosoft.liquid.LiquidLookAndFeel")); 
            menu.add(getLAFMenuItem("Next LAF v0.1", "nextlf.plaf.NextLookAndFeel")); 
            menu.add(getLAFMenuItem("Office LAFs v2.7 - Office XP", "org.fife.plaf.OfficeXP.OfficeXPLookAndFeel")); 
            menu.add(getLAFMenuItem("Office LAFs v2.7 - Office 2003", "org.fife.plaf.Office2003.Office2003LookAndFeel")); 
            menu.add(getLAFMenuItem("Office LAFs v2.7 - Visual Studio 2005", "org.fife.plaf.VisualStudio2005.VisualStudio2005LookAndFeel")); 
            menu.add(getLAFMenuItem("Oyoaha LAF v3.0", "com.oyoaha.swing.plaf.oyoaha.OyoahaLookAndFeel")); 
            menu.add(getLAFMenuItem("Substance v3.3.01 Release Japan", "org.jvnet.substance.SubstanceLookAndFeel")); 
        } 
 
        return menu; 
    } 
 
    private static JMenuItem getLAFMenuItem(String name, final String className) 
    { 
        lafNameAndClassName.put(name, className); 
 
        JMenuItem item = new JCheckBoxMenuItem(name); 
        item.setSelected(className.equals(UIManager.getCrossPlatformLookAndFeelClassName())); 
        item.addActionListener(new ActionListener() 
        { 
            public void actionPerformed(ActionEvent event) 
            { 
                applyLAF(className); 
            } 
        }); 
 
        return item; 
    } 
 
    private static void updateCheckBoxes() 
    { 
        for (int i = 0; i < menu.getItemCount(); i++) 
        { 
            JMenuItem item = menu.getItem(i); 
            if (item == null) 
            { 
                continue; // Menu separator 
            } 
            String className = lafNameAndClassName.get(item.getText()); 
            Properties props = MainFrame.getInstance().getProperties(); 
            item.setSelected(props.getProperty(PropertyManager.PROP_CURRENT_LOOK_AND_FEEL).equals(className)); 
        } 
    } 
 
    public static void applyLAF(String className) 
    { 
        try 
        { 
            // Set new LAF 
            UIManager.setLookAndFeel(className); 
 
            // Updating the components 
            for (Component listener : listeners) 
            { 
                SwingUtilities.updateComponentTreeUI(listener); 
            } 
 
            // Remember properties 
            Properties props = MainFrame.getInstance().getProperties(); 
            props.setProperty(PropertyManager.PROP_CURRENT_LOOK_AND_FEEL, className); 
            updateCheckBoxes(); 
        } 
        catch (Exception e) 
        { 
            e.printStackTrace(); 
        } 
    } 
 
    public static boolean isLAFAvailable(String className) 
    { 
        return lafNameAndClassName.values().contains(className); 
    } 
}
