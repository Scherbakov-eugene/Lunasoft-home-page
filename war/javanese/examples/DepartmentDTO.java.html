package educationalcenter.database.dto; 
 
import javax.persistence.*; 
 
@Entity 
@Table(name = "T_Department") 
public class DepartmentDTO implements Comparable<DepartmentDTO> 
{ 
    private Long departmentId; 
    private String departmentCode; 
    private String departmentShortName; 
    private String departmentFullName; 
 
    @Id 
    @GeneratedValue 
    @Column(name = "DepartmentId") 
    public Long getDepartmentId() 
    { 
        return departmentId; 
    } 
 
    public void setDepartmentId(Long departmentId) 
    { 
        this.departmentId = departmentId; 
    } 
 
    @Column(name = "DepartmentCode") 
    public String getDepartmentCode() 
    { 
        return departmentCode; 
    } 
 
    public void setDepartmentCode(String departmentCode) 
    { 
        this.departmentCode = departmentCode; 
    } 
 
    @Column(name = "DepartmentShortName") 
    public String getDepartmentShortName() 
    { 
        return departmentShortName; 
    } 
 
    public void setDepartmentShortName(String departmentShortName) 
    { 
        this.departmentShortName = departmentShortName; 
    } 
 
    @Column(name = "DepartmentFullName") 
    public String getDepartmentFullName() 
    { 
        return departmentFullName; 
    } 
 
    public void setDepartmentFullName(String departmentFullName) 
    { 
        this.departmentFullName = departmentFullName; 
    } 
 
    @Override 
    public String toString() 
    { 
        StringBuffer buffer = new StringBuffer(); 
 
        buffer.append("DepartmentDTO"); 
        buffer.append("("); 
        buffer.append("departmentId = ").append(departmentId); 
        buffer.append("; "); 
        buffer.append("departmentCode = ").append(departmentCode); 
        buffer.append("; "); 
        buffer.append("departmentShortName = ").append(departmentShortName); 
        buffer.append("; "); 
        buffer.append("departmentFullName = ").append(departmentFullName); 
        buffer.append(")"); 
 
        return buffer.toString(); 
    } 
 
    public void trimAll() 
    { 
        if (departmentCode != null) departmentCode = departmentCode.trim(); 
        if (departmentShortName != null) departmentShortName = departmentShortName.trim(); 
        if (departmentFullName != null) departmentFullName = departmentFullName.trim(); 
    } 
 
    public int compareTo(DepartmentDTO dto) 
    { 
        String name1 = this.getDepartmentCode(); 
        String name2 = dto.getDepartmentCode(); 
        return name1.compareToIgnoreCase(name2); 
    } 
 
    @Override 
    public DepartmentDTO clone() 
    { 
        DepartmentDTO dto = new DepartmentDTO(); 
 
        dto.setDepartmentId(departmentId); 
        dto.setDepartmentCode(departmentCode); 
        dto.setDepartmentShortName(departmentShortName); 
        dto.setDepartmentFullName(departmentFullName); 
 
        return dto; 
    } 
}
