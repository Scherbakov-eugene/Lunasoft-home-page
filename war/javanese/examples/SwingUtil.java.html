package yourprojectpackage.util; 
 
import java.awt.*; 
 
public class SwingUtil 
{ 
    public static void setDoubleWidth(Window window) 
    { 
        Dimension size = window.getSize(); 
        window.setSize(2 * size.width, size.height); 
    } 
 
    public static void setCentered(Window window) 
    { 
        Dimension scr = window.getToolkit().getScreenSize(); 
        Dimension wnd = window.getSize(); 
        int x = (scr.width - wnd.width) / 2; 
        int y = (scr.height - wnd.height) / 2; 
        window.setLocation(x, y); 
    } 
}
