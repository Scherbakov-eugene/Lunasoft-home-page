package educationalcenter.database.dto; 
 
import javax.persistence.*; 
 
@Entity 
@Table(name = "T_Teacher") 
public class TeacherDTO implements Comparable<TeacherDTO> 
{ 
    private Long teacherId; 
    private String teacherLastName; 
    private String teacherFirstName; 
    private String teacherPatronymic; 
    private String teacherExtStr; 
    private DepartmentDTO teacherDepartment; 
 
    @Id 
    @GeneratedValue 
    @Column(name = "TeacherId") 
    public Long getTeacherId() 
    { 
        return teacherId; 
    } 
 
    public void setTeacherId(Long teacherId) 
    { 
        this.teacherId = teacherId; 
    } 
 
    @Column(name = "TeacherLastName") 
    public String getTeacherLastName() 
    { 
        return teacherLastName; 
    } 
 
    public void setTeacherLastName(String teacherLastName) 
    { 
        this.teacherLastName = teacherLastName; 
    } 
 
    @Column(name = "TeacherFirstName") 
    public String getTeacherFirstName() 
    { 
        return teacherFirstName; 
    } 
 
    public void setTeacherFirstName(String teacherFirstName) 
    { 
        this.teacherFirstName = teacherFirstName; 
    } 
 
    @Column(name = "TeacherPatronymic") 
    public String getTeacherPatronymic() 
    { 
        return teacherPatronymic; 
    } 
 
    public void setTeacherPatronymic(String teacherPatronymic) 
    { 
        this.teacherPatronymic = teacherPatronymic; 
    } 
 
    @Column(name = "TeacherExtStr") 
    public String getTeacherExtStr() 
    { 
        return teacherExtStr; 
    } 
 
    public void setTeacherExtStr(String teacherExtStr) 
    { 
        this.teacherExtStr = teacherExtStr; 
    } 
 
    @ManyToOne 
    @JoinColumn(name = "TeacherDepartment") 
    public DepartmentDTO getTeacherDepartment() 
    { 
        return teacherDepartment; 
    } 
 
    public void setTeacherDepartment(DepartmentDTO teacherDepartment) 
    { 
        this.teacherDepartment = teacherDepartment; 
    } 
 
    @Override 
    public String toString() 
    { 
        StringBuffer buffer = new StringBuffer(); 
 
        buffer.append("TeacherDTO"); 
        buffer.append("("); 
        buffer.append("teacherId = ").append(teacherId); 
        buffer.append("; "); 
        buffer.append("teacherLastName = ").append(teacherLastName); 
        buffer.append("; "); 
        buffer.append("teacherFirstName = ").append(teacherFirstName); 
        buffer.append("; "); 
        buffer.append("teacherPatronymic = ").append(teacherPatronymic); 
        buffer.append("; "); 
        buffer.append("teacherExtStr = ").append(teacherExtStr); 
        buffer.append("; "); 
        buffer.append("teacherDepartment = ").append(teacherDepartment); 
        buffer.append(")"); 
 
        return buffer.toString(); 
    } 
 
    public void trimAll() 
    { 
        if (teacherLastName != null) teacherLastName = teacherLastName.trim(); 
        if (teacherFirstName != null) teacherFirstName = teacherFirstName.trim(); 
        if (teacherPatronymic != null) teacherPatronymic = teacherPatronymic.trim(); 
        if (teacherExtStr != null) teacherExtStr = teacherExtStr.trim(); 
    } 
 
    public int compareTo(TeacherDTO dto) 
    { 
        String name1 = this.getTeacherLastName(); 
        String name2 = dto.getTeacherLastName(); 
        return name1.compareToIgnoreCase(name2); 
    } 
 
    @Override 
    public TeacherDTO clone() 
    { 
        TeacherDTO dto = new TeacherDTO(); 
 
        dto.setTeacherId(teacherId); 
        dto.setTeacherLastName(teacherLastName); 
        dto.setTeacherFirstName(teacherFirstName); 
        dto.setTeacherPatronymic(teacherPatronymic); 
        dto.setTeacherExtStr(teacherExtStr); 
        dto.setTeacherDepartment(teacherDepartment); 
 
        return dto; 
    } 
}
