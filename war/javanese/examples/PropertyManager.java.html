package yourprojectpackage.resources; 
 
import java.io.File; 
import java.io.FileInputStream; 
import java.io.IOException; 
import java.io.FileOutputStream; 
import java.util.Properties; 
 
public class PropertyManager 
{ 
    // File names 
    public static final String FILENAME_PROPERTIES = "Settings.properties"; 
    public static final String FILENAME_MESSAGES = "Messages_unicode.properties"; 
 
    // Properties 
    public static final String PROP_1 = "prop1"; 
    public static final String PROP_2 = "prop2"; 
 
    public static void loadMessages(Properties messages) 
    { 
        File propFile = new File(FILENAME_MESSAGES); 
 
        try 
        { 
            messages.load(new FileInputStream(propFile)); 
        } 
        catch (IOException e) 
        { 
            e.printStackTrace(); 
        } 
    } 
 
    public static void loadProperties(Properties properties) 
    { 
        File propFile = new File(FILENAME_PROPERTIES); 
 
        try 
        { 
            properties.load(new FileInputStream(propFile)); 
        } 
        catch (IOException e) 
        { 
            // Properties can't loaded and empty properties will be used 
            e.printStackTrace(); 
        } 
    } 
 
    public static void saveProperties(Properties properties) 
    { 
        File propFile = new File(FILENAME_PROPERTIES); 
 
        try 
        { 
            properties.store(new FileOutputStream(propFile), null); 
        } 
        catch (IOException e) 
        { 
            // Properties can't saved and nothing to do 
            e.printStackTrace(); 
        } 
    } 
}
